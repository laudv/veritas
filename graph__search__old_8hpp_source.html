303 bool step_for(double num_seconds, size_t num_steps)
304 {
305 double start = time_since_start();
306 bool done = false;
307
308 while (!done)
309 {
310 if (stop_conditions_met())
311 break;
312
313 double dur = time_since_start() - start;
314 done = steps(num_steps);
315 if (dur >= num_seconds)
316 break;
317 }
318
319 return done;
320 }

322 bool stop_conditions_met() const
323 {
324 auto [lo, up_a, up_ara] = current_bounds_with_base_score();
325 if (std::min(up_a, up_ara) < stop_when_up_bound_less_than)
326 {
327 std::cout << "stop_conditions_met: stopping early: "
328 << "upper bound " << stop_when_up_bound_less_than
329 << " reached (" << up_a << ", " << up_ara << ")"
330 << std::endl;
331 return true;
332 }
333
334 if (solutions_.size() > 0 && solutions_[0].eps == stop_when_solution_eps_equals)
335 {
336 std::cout << "stop_conditions_met: stopping early: "
337 << "solution with eps " << stop_when_solution_eps_equals
338 << " found." << std::endl;
339 return true;
340 }
341
342 if (solutions_.size() >= stop_when_num_solutions_equals)
343 {
344 std::cout << "stop_conditions_met: stopping early: "
345 << num_solutions() << " >= "
346 << stop_when_num_solutions_equals
347 << " solutions found"
348 << std::endl;
349 return true;
350 }
351
352 if (solutions_.size() > 0 && states_[solutions_.front().state_index].fscore()
353 > stop_when_solution_output_greater_than)
354 {
355 std::cout << "stop_conditions_met: stopping early: "
356 << "solution found with output greater than "
357 << stop_when_solution_output_greater_than
358 << std::endl;
359 return true;
360 }
361
362 return false;
363 }

585 void expand(size_t state_index)
586 {
587 find_indep_sets_not_added(state_index, workspace_.indep_sets);
588
589 states_[state_index].is_expanded = true;
590 BoxRef parent_box = states_[state_index].box;
591 int next_indep_set = states_[state_index].next_indep_set;
592
593 // expand by adding vertexes from `next_indep_set`
594 Graph::IndepSet set = g_.get_vertices(next_indep_set);
595 int num_vertices = static_cast<int>(set.size());
596 for (int vertex = 0; vertex < num_vertices; ++vertex)
597 {
598 const Graph::Vertex& v = set[vertex];
599 if (v.box.overlaps(parent_box))
600 {
601 combine_boxes(v.box, parent_box, true, workspace_.box);
602 construct_and_push_states(state_index, v);
603 }
604 }

713 // assume valid state for parent state in `workspace_.indep_sets`
714 void compute_heuristic_dynprog(State& new_state, int skip_indep_set) const
715 {
716 std::vector<FloatT> d0, d1;
717 FloatT maxmax = -FLOATT_INF;
718
719 // fill d0 with first indep_set's output values
720 int prev_indep_set = -1;
721
722 // compute heuristic using dynamic programming
723 for (size_t i = 0; i < workspace_.indep_sets.size(); ++i)
724 {
725 FloatT tree_max = -FLOATT_INF;
726
727 int indep_set = workspace_.indep_sets[i];
728 if (indep_set == skip_indep_set)
729 continue;
730
731 // fill d0 with the first indep_set
732 if (d0.empty())
733 {
734 for (const auto& v : g_.get_vertices(indep_set))
735 {
736 if (v.box.overlaps(new_state.box))
737 {
738 d0.push_back(v.output);
739 tree_max = std::max(tree_max, v.output);
740 }
741 else
742 {
743 d0.push_back(-FLOATT_INF);
744 }
745 }
746 }
747 else
748 {
749 const auto& set0 = g_.get_vertices(prev_indep_set);
750 const auto& set1 = g_.get_vertices(indep_set);
751 for (const auto& v1 : set1)
752 {
753 FloatT max = -FLOATT_INF;
754
755 if (v1.box.overlaps(new_state.box))
756 {
757 tree_max = std::max(tree_max, v1.output);
758 for (size_t j = 0; j < d0.size(); ++j)
759 {
760 const auto& v0 = set0[j];
761 if (v0.box.overlaps(v1.box) && max < d0[j])
762 max = d0[j];
763 }
764 d1.push_back(max + v1.output);
765 }
766 else
767 {
768 d1.push_back(-FLOATT_INF);
769 }
770 }
771
772 d0.clear();
773 std::swap(d0, d1);
774 }
775
776 if (maxmax < tree_max)
777 {
778 maxmax = tree_max;
779 new_state.next_indep_set = indep_set;
780 }
781
782 prev_indep_set = indep_set;
783 }
784
785 // find max of d0 --> thats the heuristic value
786 if (new_state.next_indep_set != -1)
787 {
788 FloatT max = -FLOATT_INF;
789 for (auto v : d0)
790 max = (v > max) ? v : max;
791 new_state.h = max;
792 }
793 else // this is a final state with no heuristic estimate
794 {
795 new_state.h = 0.0;
796 }
797 }
